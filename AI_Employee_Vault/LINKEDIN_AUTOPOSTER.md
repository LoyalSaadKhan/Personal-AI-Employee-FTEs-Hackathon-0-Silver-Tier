# LinkedIn Auto-Poster - Silver Tier

## Overview

This skill automatically generates and posts content to LinkedIn for:
- Lead generation
- Brand building
- Sharing expertise
- Engaging with audience

All posts go through HITL approval before publishing (configurable).

---

## Configuration

### LinkedIn Credentials

**Option 1: LinkedIn API (Recommended for production)**

1. Go to [LinkedIn Developer Portal](https://www.linkedin.com/developers/)
2. Create new app
3. Get Client ID and Client Secret
4. Request `w_member_social` permission
5. Save credentials to `~/.linkedin/credentials.json`

**Option 2: Browser Automation (For personal use)**

Uses Playwright to automate LinkedIn web interface.
Session saved in `~/.linkedin/session`

### Configuration File

```json
// ~/.qwen/linkedin_config.json
{
  "post_frequency": "daily",
  "post_time": "09:00",
  "timezone": "UTC",
  "content_types": ["tip", "announcement", "question", "case_study", "insight"],
  "hashtags": [
    "#AI",
    "#Automation",
    "#Productivity",
    "#DigitalTransformation",
    "#BusinessEfficiency"
  ],
  "require_approval": true,
  "auto_approve_themes": ["tips", "industry_insights"],
  "max_posts_per_day": 2
}
```

## Implementation

### LinkedIn Poster Script

```python
#!/usr/bin/env python3
"""
LinkedIn Auto-Poster - Generates and posts content to LinkedIn.

Usage:
    python linkedin_poster.py /path/to/vault --draft    # Generate draft
    python linkedin_poster.py /path/to/vault --publish  # Publish scheduled
    python linkedin_poster.py /path/to/vault --schedule # Schedule for later
"""

import json
import random
from pathlib import Path
from datetime import datetime, timedelta

class LinkedInPoster:
    def __init__(self, vault_path: str):
        self.vault_path = Path(vault_path)
        self.social_folder = self.vault_path / 'Social'
        self.drafts_folder = self.social_folder / 'drafts'
        self.scheduled_folder = self.social_folder / 'scheduled'
        self.published_folder = self.social_folder / 'published'
        
        # Ensure folders exist
        for folder in [self.social_folder, self.drafts_folder, 
                       self.scheduled_folder, self.published_folder]:
            folder.mkdir(parents=True, exist_ok=True)
        
        # Load config
        self.config = self._load_config()
    
    def _load_config(self) -> dict:
        """Load LinkedIn configuration."""
        config_path = Path.home() / '.qwen' / 'linkedin_config.json'
        if config_path.exists():
            return json.loads(config_path.read_text())
        
        # Default config
        return {
            'post_frequency': 'daily',
            'post_time': '09:00',
            'content_types': ['tip', 'question', 'insight'],
            'hashtags': ['#AI', '#Automation', '#Productivity'],
            'require_approval': True,
            'auto_approve_themes': ['tips']
        }
    
    def generate_post(self) -> Path:
        """Generate a new LinkedIn post draft."""
        # Select random content type
        content_type = random.choice(self.config['content_types'])
        
        # Generate content based on type
        if content_type == 'tip':
            content = self._generate_tip_post()
        elif content_type == 'question':
            content = self._generate_question_post()
        elif content_type == 'insight':
            content = self._generate_insight_post()
        else:
            content = self._generate_generic_post()
        
        # Add hashtags
        hashtags = ' '.join(random.sample(self.config['hashtags'], 
                                          min(3, len(self.config['hashtags']))))
        content['hashtags'] = hashtags
        
        # Create draft file
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        filename = f"LINKEDIN_post_{timestamp}.md"
        filepath = self.drafts_folder / filename
        
        # Determine if auto-approve eligible
        auto_approve = content_type in self.config.get('auto_approve_themes', [])
        
        draft_content = f'''---
type: social_post
platform: linkedin
created: {datetime.now().isoformat()}
scheduled: {(datetime.now() + timedelta(hours=1)).isoformat()}
status: draft
content_type: {content_type}
auto_approve: {str(auto_approve).lower()}
---

# LinkedIn Post Draft

## Content

{content['text']}

## Hashtags
{content['hashtags']}

## Approval Required
{"Auto-approve eligible" if auto_approve else "Yes - Move to /Approved to publish"}

---

*Generated by AI Employee v0.1 (Silver Tier)*
'''
        filepath.write_text(draft_content)
        
        return filepath
    
    def _generate_tip_post(self) -> dict:
        """Generate a tip-style post."""
        tips = [
            {
                'hook': '🚀 Quick Tip:',
                'body': 'Did you know the average professional spends 2.5 hours daily on repetitive tasks?',
                'list': ['✅ Email responses', '✅ Meeting scheduling', '✅ Invoice generation', '✅ Social media posting'],
                'cta': 'Start small. Pick ONE task. Automate it. Reclaim your time.',
                'question': "What's the first task you'd automate? Drop a comment! 👇"
            },
            # Add more templates...
        ]
        
        tip = random.choice(tips)
        text = f"""{tip['hook']}

{tip['body']}

Here's what you can automate TODAY:
{chr(10).join(tip['list'])}

{tip['cta']}

{tip['question']}"""
        
        return {'text': text, 'type': 'tip'}
    
    def _generate_question_post(self) -> dict:
        """Generate a question-style post."""
        questions = [
            "What's your biggest challenge with automation?",
            "How many hours per week do you spend on repetitive tasks?",
            "What tool has transformed your productivity the most?",
            "What's one task you wish you could automate but haven't yet?"
        ]
        
        text = f"""💬 Let's Discuss:

{random.choice(questions)}

I'm curious to hear your experiences and learn from the community.

Share your thoughts in the comments! 👇"""
        
        return {'text': text, 'type': 'question'}
    
    def _generate_insight_post(self) -> dict:
        """Generate an insight-style post."""
        insights = [
            "Here's what I learned about AI automation after helping 50+ businesses...",
            "The biggest misconception about automation is that it replaces humans. In reality...",
            "After analyzing 1000+ automated workflows, here's the pattern I noticed..."
        ]
        
        text = f"""💡 Industry Insight:

{random.choice(insights)}

[Add your unique perspective and value here]

What's your take on this?"""
        
        return {'text': text, 'type': 'insight'}
    
    def _generate_generic_post(self) -> dict:
        """Generate a generic post."""
        return {
            'text': "📝 Update from AI Employee\n\nWorking on exciting automation projects today!\n\n#Productivity #AI",
            'type': 'generic'
        }
    
    def publish_post(self, post_path: Path) -> bool:
        """Publish a scheduled post to LinkedIn."""
        # Implementation depends on LinkedIn API or browser automation
        # For now, log the action
        content = post_path.read_text()
        
        # Log publication
        log_entry = {
            'timestamp': datetime.now().isoformat(),
            'action': 'linkedin_post_published',
            'file': post_path.name,
            'status': 'success'
        }
        
        log_file = self.published_folder / f'{datetime.now().strftime("%Y-%m-%d")}.json'
        if log_file.exists():
            logs = json.loads(log_file.read_text())
        else:
            logs = []
        logs.append(log_entry)
        log_file.write_text(json.dumps(logs, indent=2))
        
        return True


def main():
    import argparse
    
    parser = argparse.ArgumentParser(description='LinkedIn Auto-Poster')
    parser.add_argument('vault_path', help='Path to Obsidian vault')
    parser.add_argument('--draft', action='store_true', help='Generate draft post')
    parser.add_argument('--publish', action='store_true', help='Publish scheduled posts')
    
    args = parser.parse_args()
    
    poster = LinkedInPoster(args.vault_path)
    
    if args.draft:
        filepath = poster.generate_post()
        print(f'Draft created: {filepath}')
    elif args.publish:
        # Publish all scheduled posts
        for post_file in poster.scheduled_folder.glob('*.md'):
            poster.publish_post(post_file)
            print(f'Published: {post_file.name}')


if __name__ == '__main__':
    main()
```

## Best Practices

### Content Guidelines

1. **Keep it concise** - 3-5 short paragraphs max
2. **Use emojis sparingly** - 2-4 per post
3. **Include CTA** - Ask a question or invite comments
4. **Add hashtags** - 3-5 relevant hashtags
5. **Post consistently** - Same time daily for algorithm

### Approval Guidelines

1. **Auto-approve**: Tips, industry insights, general motivation
2. **Require approval**: Client mentions, specific numbers, controversial topics
3. **Never auto-post**: Political content, sensitive topics, unverified claims
